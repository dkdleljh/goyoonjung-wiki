```yaml
    - tvn.com # 방송사
    - netflix.com # OTT
    - award_sites # 시상식

  cross_validation:
    - compare_multiple_sources
    - check_date_consistency
    - verify_official_announcements

  auto_flagging:
    - missing_sources
    - outdated_info
    - contradictory_data

  quality_score:
    - completeness_score
    - accuracy_score
    - freshness_score
```
